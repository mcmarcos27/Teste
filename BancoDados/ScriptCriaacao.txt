create table MARCOS.CLIENTES (
	ID_CLIENTE  INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
    NOME VARCHAR(300) NOT NULL,
    CIDADE VARCHAR(300),
    UF    VARCHAR(2)
);

create table MARCOS.PRODUTOS (
	ID_PRODUTO  INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
    DESCRICAO VARCHAR(300) NOT NULL,
    PRECO numeric(12,2) DEFAULT 0 NOT NULL
);

CREATE TABLE MARCOS.PEDIDOS (
	NUMERO_PED  INT NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    DTEMISSAO DATE   NOT NULL,
	ID_CLIENTE INT NOT NULL,
    VLTOTAL numeric(12,2) DEFAULT 0 NOT NULL    
);
ALTER TABLE MARCOS.PEDIDOS ADD CONSTRAINT FK_PEDIDO_CLIENTE foreign key (ID_CLIENTE) REFERENCES MARCOS.CLIENTES (ID_CLIENTE);
CREATE TABLE MARCOS.ITENSPEDIDO (
	ID_ITEM  INT NOT NULL PRIMARY KEY auto_increment,
    NUMERO_PED  INT NOT NULL ,
	ID_PRODUTO INT NOT NULL,
    QUANTIDADE numeric(12,2) DEFAULT 0 NOT NULL,
    VALOR_UNITARIO numeric(12,2) DEFAULT 0 NOT NULL,    
	VLTOTAL numeric(12,2) DEFAULT 0 NOT NULL    
);
ALTER TABLE MARCOS.ITENSPEDIDO ADD CONSTRAINT FK_ITENSPEDIDO_PEDIDOS foreign key (NUMERO_PED) REFERENCES MARCOS.PEDIDOS (NUMERO_PED);
ALTER TABLE MARCOS.ITENSPEDIDO ADD CONSTRAINT FK_ITENSPEDIDO_PRODUTOS foreign key (ID_PRODUTO) REFERENCES MARCOS.PRODUTOS (ID_PRODUTO);
